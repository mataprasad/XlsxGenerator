using System;
using System.Collections.Generic;
using System.Data;
using System.Data.SqlClient;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

public partial class Default : System.Web.UI.Page
{
    protected void Page_Load(object sender, EventArgs e)
    {
        DataSet ds = get_data();
        List<string> columns = new List<string>();
        columns.Add("col1");
        columns.Add("col2");
        columns.Add("col3");
        Response.BinaryWrite(XlsxGenerator.GeneratorWeb.Generate(columns, ds.Tables[0], true));
        Response.ContentType = "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet";
        Response.AddHeader("content-disposition", string.Format("attachment;  filename={0}.xlsx",Guid.NewGuid().ToString()));
    }

    public DataSet get_data()
    {
        SqlConnection CONN = null;
        SqlCommand COMM = null;
        SqlDataAdapter ADPT = null;
        DataSet DS = null;
        try
        {
            CONN = new SqlConnection();
            CONN.ConnectionString = "<CONNECTION-STRING>";
            CONN.Open();
            COMM = new SqlCommand();
            COMM.CommandTimeout = 0;
            COMM.Connection = CONN;
            COMM.CommandType = CommandType.Text;
            COMM.CommandText = "<SELECT-QUERY>";
            ADPT = new SqlDataAdapter(COMM);
            DS = new DataSet();
            ADPT.Fill(DS);
            if (DS != null && DS.Tables.Count > 0)
            {
                return DS;
            }
        }
        catch (Exception ex)
        {
            DS = null;
            ADPT = null;
            COMM = null;
        }
        finally
        {
            CONN.Close();
        }

        return null;
    }
}
